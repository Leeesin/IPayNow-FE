一般是由于 在功能函数里面 调用了wx.login刷新登录状态 导致session过期

//html
<button  open-type="getUserInfo" bindgetuserinfo="bindGetUserInfo">授权登录</button>

//js
Page({
    bindGetUserInfo(){
        /*  错误示例 应该事先获取code
            wx.login({
                success({ code }){

                }
            })
        */
    }
});
