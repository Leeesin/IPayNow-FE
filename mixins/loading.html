# 控制 界面 loading效果

//采用约定优于配置的原则 使用时 必须 保证loading 的变量 存放到 组件data函数的 loading对象内

const setLoading = {
    methods: {
        async setLoading(field = "", key = "", fn = new Promise()) {
            this.$set(this.loading[field], key, true);
            await fn();
            this.$set(this.loading[field], key, false);
        },
    }
}
export default setLoading
